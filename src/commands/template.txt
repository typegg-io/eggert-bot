from discord.ext import commands

from bot_setup import BotContext
from commands.base import Command
from utils.messages import Page, Message, Field

info = {
    "name": "",
    "aliases": [],
    "description": "",
    "parameters": "[username]",
    "examples": [""],
}


class CommandName(Command):
    # Declare which flags this command can use, omit the line if none are supported.
    # Options: "metric", "raw", "gamemode", "status", "language", "number", "number_range", "date", "quote_id"
    supported_flags = {}

    @commands.command(aliases=info["aliases"])
    async def commandname(self, ctx: BotContext, *args: str):
        profile = await self.get_profile(ctx, args[0] if args else None)
        await run(ctx, profile)


async def run(ctx: BotContext, profile: dict):
    page = Page(
        title="Title",
        description="Description",
    )

    message = Message(
        ctx,
        page=page,
        profile=profile,
    )

    await message.send()
